class FinancialModelGenerator:
    def __init__(self):
        self.wb = Workbook()
        self.setup_styles()
        
    def setup_styles(self):
        """إعداد أنماط التنسيق"""
        self.title_font = Font(name='Arial', size=14, bold=True, color='FFFFFF')
        self.header_font = Font(name='Arial', size=10, bold=True, color='FFFFFF')
        self.normal_font = Font(name='Arial', size=10)
        self.bold_font = Font(name='Arial', size=10, bold=True)
        
        self.title_fill = PatternFill(start_color='1F4E79', end_color='1F4E79', fill_type='solid')
        self.header_fill = PatternFill(start_color='2E75B6', end_color='2E75B6', fill_type='solid')
        self.light_blue_fill = PatternFill(start_color='BDD7EE', end_color='BDD7EE', fill_type='solid')
        self.green_fill = PatternFill(start_color='C6EFCE', end_color='C6EFCE', fill_type='solid')
        self.yellow_fill = PatternFill(start_color='FFEB9C', end_color='FFEB9C', fill_type='solid')
        self.red_fill = PatternFill(start_color='FFC7CE', end_color='FFC7CE', fill_type='solid')
        
        self.thin_border = Border(
            left=Side(style='thin'), right=Side(style='thin'),
            top=Side(style='thin'), bottom=Side(style='thin')
        )
    
    def apply_cell_style(self, cell, font=None, fill=None, alignment=None, 
                        number_format=None, border=None):
        """تطبيق التنسيق على الخلية"""
        if font:
            cell.font = font
        if fill:
            cell.fill = fill
        if alignment:
            cell.alignment = alignment
        else:
            cell.alignment = Alignment(horizontal='center', vertical='center', wrap_text=True)
        if number_format:
            cell.number_format = number_format
        if border:
            cell.border = border
        else:
            cell.border = self.thin_border

    def create_cash_flow(self):
        """إنشاء صفحة التدفقات النقدية"""
        ws = self.wb.create_sheet("التدفقات النقدية و NPV")
        ws.sheet_properties.tabColor = "548235"
        
        ws.merge_cells('A1:D1')
        ws['A1'] = 'قائمة التدفقات النقدية وتحليل NPV و IRR'
        self.apply_cell_style(ws['A1'], self.title_font, self.title_fill)
        
        headers = ['السنة', 'التدفق النقدي (ريال)', 'عامل الخصم (12%)', 'القيمة الحالية (ريال)']
        for col, header in enumerate(headers, 1):
            self.apply_cell_style(ws.cell(row=3, column=col, value=header), 
                                self.header_font, self.header_fill)
        
        investment = 12226000000
        cash_flow_years = [2403000000] * 10
        salvage_value = 454000000
        discount_rate = 0.12
        
        ws.cell(row=4, column=1, value=0)
        ws.cell(row=4, column=2, value=-investment)
        ws.cell(row=4, column=3, value=1)
        ws.cell(row=4, column=4, value=-investment)
        for col in range(1, 5):
            self.apply_cell_style(ws.cell(row=4, column=col), 
                                self.normal_font, self.yellow_fill,
                                number_format='#,##0' if col in [2, 4] else None)
        
        for i, cf in enumerate(cash_flow_years, 1):
            row = 4 + i
            factor = 1 / ((1 + discount_rate) ** i)
            pv = cf * factor
            ws.cell(row=row, column=1, value=i)
            ws.cell(row=row, column=2, value=cf)
            ws.cell(row=row, column=3, value=factor)
            ws.cell(row=row, column=4, value=pv)
            for col in range(1, 5):
                fill = self.green_fill if pv > 0 else self.red_fill
                self.apply_cell_style(ws.cell(row=row, column=col), 
                                    self.normal_font, fill,
                                    number_format='#,##0.00' if col in [3, 4] else '#,##0')
        
        # سنة 10 مع الخردة
        row = 14
        cf_10 = cash_flow_years[9] + salvage_value
        factor_10 = 1 / ((1 + discount_rate) ** 10)
        pv_10 = cf_10 * factor_10
        ws.cell(row=row, column=1, value=10)
        ws.cell(row=row, column=2, value=cf_10)
        ws.cell(row=row, column=3, value=factor_10)
        ws.cell(row=row, column=4, value=pv_10)
        for col in range(1, 5):
            self.apply_cell_style(ws.cell(row=row, column=col), 
                                self.normal_font, self.green_fill,
                                number_format='#,##0.00' if col in [3, 4] else '#,##0')
        
        # المعادلات
        npv_row = 16
        ws.merge_cells(start_row=npv_row, start_column=1, end_row=npv_row, end_column=2)
        ws.cell(row=npv_row, column=1, value='صافي القيمة الحالية (NPV)')
        self.apply_cell_style(ws.cell(row=npv_row, column=1), self.bold_font, self.light_blue_fill)
        ws.cell(row=npv_row, column=3, value='=NPV(0.12,D5:D14)+D4')
        self.apply_cell_style(ws.cell(row=npv_row, column=3), self.bold_font, self.green_fill,
                            number_format='#,##0')
        
        irr_row = 17
        ws.merge_cells(start_row=irr_row, start_column=1, end_row=irr_row, end_column=2)
        ws.cell(row=irr_row, column=1, value='معدل العائد الداخلي (IRR)')
        self.apply_cell_style(ws.cell(row=irr_row, column=1), self.bold_font, self.light_blue_fill)
        ws.cell(row=irr_row, column=3, value='=IRR(D4:D14, 0.1)')
        self.apply_cell_style(ws.cell(row=irr_row, column=3), self.bold_font, self.green_fill,
                            number_format='0.00%')
        
        payback_row = 18
        ws.merge_cells(start_row=payback_row, start_column=1, end_row=payback_row, end_column=2)
        ws.cell(row=payback_row, column=1, value='فترة الاسترداد (Payback)')
        self.apply_cell_style(ws.cell(row=payback_row, column=1), self.bold_font, self.light_blue_fill)
        ws.cell(row=payback_row, column=3, value='=4+(ABS(SUM($D$4:$D$8))/D9)')
        self.apply_cell_style(ws.cell(row=payback_row, column=3), self.bold_font, self.yellow_fill,
                            number_format='0.00')
        
        return ws
    
    def create_sensitivity_analysis(self):
        """إنشاء صفحة تحليل الحساسية"""
        ws = self.wb.create_sheet("تحليل الحساسية")
        ws.sheet_properties.tabColor = "BF8F00"
        
        ws.merge_cells('A1:E1')
        ws['A1'] = 'تحليل حساسية صافي الربح (للتغيرات في المتغيرات الحرجة)'
        self.apply_cell_style(ws['A1'], self.title_font, self.title_fill)
        
        base_profit = 1731900000
        scenarios = [
            ('الحالة الأساسية (Base Case)', 0, base_profit),
            ('ارتفاع تكلفة الأعلاف (+20%)', -0.2, base_profit * 0.7),
            ('انخفاض سعر بيع الدجاج والبيض (-10%)', -0.1, base_profit * 0.4),
            ('ارتفاع سعر الصرف (تدهور 20%)', -0.2, base_profit * 0.65),
            ('ارتفاع معدل النفوق (+5%)', -0.05, base_profit * 0.85),
        ]
        
        headers = ['السيناريو', 'نسبة التغير', 'صافي الربح المتوقع (ريال)', 'التأثير', 'ملاحظات']
        for col, header in enumerate(headers, 1):
            self.apply_cell_style(ws.cell(row=3, column=col, value=header), 
                                self.header_font, self.header_fill)
        
        for i, (name, change, profit) in enumerate(scenarios, 4):
            impact = "إيجابي" if profit > base_profit else ("حرج" if profit < base_profit * 0.6 else "مقبول")
            fill = self.green_fill if impact == "إيجابي" else (self.red_fill if impact == "حرج" else self.yellow_fill)
            
            ws.cell(row=i, column=1, value=name)
            ws.cell(row=i, column=2, value=f'{change*100}%' if change != 0 else '0%')
            ws.cell(row=i, column=3, value=profit)
            ws.cell(row=i, column=4, value=impact)
            ws.cell(row=i, column=5, value='')
            for col in range(1, 6):
                self.apply_cell_style(ws.cell(row=i, column=col), 
                                    self.normal_font, fill,
                                    number_format='#,##0' if col == 3 else None)
        
        return ws

    def generate(self, filename=None):
        """توليد الملف الكامل"""
        # إنشاء جميع الصفحات
        self.create_executive_summary()
        self.create_balance_sheet()
        self.create_cash_flow()
        self.create_sensitivity_analysis()
        self.create_investment_costs()
        
        # ضبط عرض الأعمدة
        for ws in self.wb.worksheets:
            for col in range(1, 7):
                ws.column_dimensions[get_column_letter(col)].width = 25
        
        # حفظ الملف
        if filename is None:
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            filename = f'نموذج_دراسة_الجدوى_مزرعة_الدواجن_{timestamp}.xlsx'
        
        self.wb.save(filename)
        return filename
